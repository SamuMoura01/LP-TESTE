function fmtBRL(n){
  if(n>=1000000)return'R$ '+(n/1000000).toFixed(1).replace('.',',')+'M';
  if(n>=1000)return'R$ '+(n/1000).toFixed(1).replace('.',',')+'K';
  return'R$ '+Math.round(n).toLocaleString('pt-BR');
}
function calcROI(){
  var meses=+document.getElementById('s-tempo').value;
  var verba=+document.getElementById('s-verba').value;
  var ticket=+document.getElementById('s-ticket').value;
  var conv=+document.getElementById('s-conv').value/100;

  document.getElementById('v-tempo').textContent=meses+(meses===1?' mês':' meses');
  document.getElementById('v-verba').textContent='R$ '+verba.toLocaleString('pt-BR');
  document.getElementById('v-ticket').textContent='R$ '+ticket.toLocaleString('pt-BR');
  document.getElementById('v-conv').textContent=Math.round(conv*100)+'%';

  var otimFator=1+Math.min((meses-3)*0.04,0.6);
  var totalVerba=verba*meses;
  var cpl=2;
  var leads=Math.round((totalVerba/cpl)*otimFator);
  var vendas=Math.round(leads*conv);
  var faturamento=vendas*ticket;
  var roi=totalVerba>0?((faturamento-totalVerba)/totalVerba*100):0;
  var orgX=Math.max(2,meses*0.35);

  document.getElementById('r-leads').textContent=leads.toLocaleString('pt-BR');
  document.getElementById('r-vendas').textContent=vendas.toLocaleString('pt-BR');
  document.getElementById('r-fat').textContent=fmtBRL(faturamento);
  document.getElementById('r-roi').textContent=roi>0?'+'+Math.
